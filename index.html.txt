/* ============ Sync with Google Sheets ============ */
function setSyncStatus(s){
  const icon = document.getElementById('syncStatusIcon');
  const text = document.getElementById('syncStatusText');
  if(!icon || !text) return;
  if(!syncUrl){ icon.textContent='☁️'; text.textContent='Sheets'; return; }
  switch(s){
    case 'pending':  icon.textContent='⏳'; text.textContent='รอ sync'; break;
    case 'syncing':  icon.textContent='🔄'; text.textContent='กำลัง sync'; break;
    case 'success':  icon.textContent='✅'; text.textContent='sync แล้ว'; break;
    case 'error':    icon.textContent='❌'; text.textContent='ผิดพลาด'; break;
    default:         icon.textContent='☁️'; text.textContent= autoSync?'auto-sync':'manual';
  }
}

function scheduleSync(){
  if(!syncUrl){ return; }
  setSyncStatus('pending');
  clearTimeout(syncDebounce);
  syncDebounce = setTimeout(pushToSheet, 1800);
}

async function pushToSheet(silent){
  if(!syncUrl){ if(!silent) alert('ยังไม่ได้ตั้งค่า Web App URL'); return; }
  if(syncing) return;
  syncing = true;
  setSyncStatus('syncing');
  try{
    const resp = await fetch(syncUrl, {
      method:'POST',
      body: JSON.stringify({action:'push', data: state}),
      headers: {'Content-Type':'text/plain;charset=utf-8'},
      redirect: 'follow'
    });
    const result = await resp.json();
    if(result.ok){
      setSyncStatus('success');
      if(!silent) toast('☁️ '+(result.message||'sync สำเร็จ'));
      setTimeout(()=>setSyncStatus('idle'), 2200);
    } else {
      throw new Error(result.error||'ไม่ทราบสาเหตุ');
    }
  }catch(err){
    setSyncStatus('error');
    if(!silent) alert('Sync ไม่สำเร็จ: '+err.message+'\n\nตรวจสอบว่า URL ถูกต้องและ Deploy เป็น "ทุกคน" (Anyone)');
    console.error(err);
  }finally{
    syncing = false;
  }
}

async function pullFromSheet(){
  if(!syncUrl){ alert('ยังไม่ได้ตั้งค่า Web App URL — กดปุ่ม ☁️ Sheets เพื่อตั้งค่า'); return; }
  if(!confirm('⚠️ ดึงข้อมูลจาก Google Sheet มาแทนข้อมูลปัจจุบัน?\n(ข้อมูลในแอพปัจจุบันจะถูกแทนที่)')) return;
  syncing = true;
  setSyncStatus('syncing');
  try{
    const resp = await fetch(syncUrl+'?action=pull', {redirect:'follow'});
    const result = await resp.json();
    if(result.ok && result.data){
      // ถ้า Sheet ยังว่าง ให้คงข้อมูลปัจจุบันไว้
      if(!result.data.communities || result.data.communities.length===0){
        toast('Sheet ยังไม่มีข้อมูล — คงข้อมูลปัจจุบันไว้');
      } else {
        state = result.data;
        currentCommunityId = state.communities[0]?.id;
        localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
        renderAll();
        toast('✅ ดึงข้อมูล '+result.data.communities.length+' หมู่บ้านจาก Sheet แล้ว');
      }
      setSyncStatus('success');
      setTimeout(()=>setSyncStatus('idle'), 2200);
    } else {
      throw new Error(result.error||'ไม่ทราบสาเหตุ');
    }
  }catch(err){
    setSyncStatus('error');
    alert('ดึงข้อมูลไม่สำเร็จ: '+err.message);
  }finally{
    syncing = false;
  }
}

  const c = currentCommunity();
  if(!c) return;

    }
  };

function svgPoint(svg, evt){
  const rect = svg.getBoundingClientRect();
  const vb = svg.viewBox.baseVal;
  const x = (evt.clientX - rect.left) * (vb.width / rect.width);
  const y = (evt.clientY - rect.top) * (vb.height / rect.height);
  return {x:Math.round(x), y:Math.round(y)};
}

function drawMapPoint(p){
  const svg = document.getElementById('mapCanvas');
  const color = TYPE_COLORS[p.type] || '#64748b';
  const g = document.createElementNS('http://www.w3.org/2000/svg','g');
  g.setAttribute('class','map-point-group');
  g.setAttribute('data-id', p.id);
  g.setAttribute('transform',`translate(${p.x},${p.y})`);
  g.style.cursor='pointer';

  // Shadow
  const shadow = document.createElementNS('http://www.w3.org/2000/svg','circle');
  shadow.setAttribute('cx',0); shadow.setAttribute('cy',2);
  shadow.setAttribute('r',12); shadow.setAttribute('fill','rgba(0,0,0,.15)');
  g.appendChild(shadow);

  // Main dot
  const dot = document.createElementNS('http://www.w3.org/2000/svg','circle');
  dot.setAttribute('cx',0); dot.setAttribute('cy',0);
  dot.setAttribute('r',12); dot.setAttribute('fill',color);
  dot.setAttribute('stroke','#fff'); dot.setAttribute('stroke-width',2);
  g.appendChild(dot);

  // Icon
  const txt = document.createElementNS('http://www.w3.org/2000/svg','text');
  txt.setAttribute('x',0); txt.setAttribute('y',5);
  txt.setAttribute('text-anchor','middle');
  txt.setAttribute('font-size','14');
  txt.textContent = TYPE_ICONS[p.type]||'•';
  g.appendChild(txt);

  // Label
  if(p.name){
    const label = document.createElementNS('http://www.w3.org/2000/svg','text');
    label.setAttribute('x',0); label.setAttribute('y',-18);
    label.setAttribute('text-anchor','middle');
    label.setAttribute('font-size','11');
    label.setAttribute('fill','#0f172a');
    label.setAttribute('font-family','inherit');
    label.setAttribute('font-weight','600');
    label.textContent = p.name.length>20 ? p.name.slice(0,20)+'…' : p.name;
    g.appendChild(label);
  }

  g.onclick = (e)=>{
    e.stopPropagation();
    if(mapMode==='delete'){
      if(confirm('ลบจุด "'+p.name+'" ?')){
        const c = currentCommunity();
        c.mapPoints = c.mapPoints.filter(x=>x.id!==p.id);
        saveState(); renderMap();
      }
    }else if(mapMode==='move'){
      svg.style.cursor='move';
      const move = (ev)=>{
        const pt = svgPoint(svg, ev);
        g.setAttribute('transform',`translate(${pt.x},${pt.y})`);
        p.x = pt.x; p.y = pt.y;
      };
      const up = ()=>{
        document.removeEventListener('mousemove',move);
        document.removeEventListener('mouseup',up);
        svg.style.cursor='crosshair';
        saveState();
      };
      document.addEventListener('mousemove',move);
      document.addEventListener('mouseup',up);
    }else{
      openMapModal(p);
    }
  };
  svg.appendChild(g);
}

function renderGenoChart(){
  const c = currentCommunity();
  const svg = document.getElementById('genoCanvas');
  svg.innerHTML = '';
  if(!c.persons.length){
    svg.innerHTML = '<text x="500" y="300" text-anchor="middle" fill="#94a3b8" font-size="16" font-family="inherit">ยังไม่มีบุคคล — กดปุ่ม "เพิ่มบุคคล" เพื่อเริ่มต้น</text>';
    return;
  }

  // Build hierarchy by generations
  // Generation logic: roots (no parent) = gen 0, children = gen of parent + 1
  const persons = c.persons;
  const genMap = {};
  function getGen(p, visited=new Set()){
    if(genMap[p.id]!==undefined) return genMap[p.id];
    if(visited.has(p.id)) return 0;
    visited.add(p.id);
    if(!p.parentId){ genMap[p.id]=0; return 0; }
    const parent = persons.find(x=>x.id===p.parentId);
    if(!parent){ genMap[p.id]=0; return 0; }
    const g = getGen(parent, visited)+1;
    genMap[p.id] = g;
    return g;
  }
  persons.forEach(p=>getGen(p));
  // Group by generation
  const gens = {};
  persons.forEach(p=>{
    const g = genMap[p.id];
    (gens[g] = gens[g]||[]).push(p);
  });

  const W = 1000, H = 600;
  const gKeys = Object.keys(gens).map(Number).sort((a,b)=>a-b);
  const rowH = Math.min(150, (H-80)/Math.max(gKeys.length,1));

  const pos = {};
  gKeys.forEach((g,gi)=>{
    const arr = gens[g];
    const spacing = W/(arr.length+1);
    arr.forEach((p,i)=>{
      pos[p.id] = { x: spacing*(i+1), y: 60 + gi*rowH };
    });
  });

  // Draw connections (parent->child)
  persons.forEach(p=>{
    if(p.parentId && pos[p.parentId] && pos[p.id]){
      const a = pos[p.parentId], b = pos[p.id];
      const line = document.createElementNS('http://www.w3.org/2000/svg','path');
      line.setAttribute('d',`M ${a.x} ${a.y+25} L ${a.x} ${(a.y+b.y)/2} L ${b.x} ${(a.y+b.y)/2} L ${b.x} ${b.y-25}`);
      line.setAttribute('stroke','#64748b'); line.setAttribute('fill','none'); line.setAttribute('stroke-width','2');
      svg.appendChild(line);
    }
  });

  // Draw spouse lines (double line)
  persons.forEach(p=>{
    if(p.spouseId && pos[p.spouseId] && pos[p.id] && p.id < p.spouseId){
      const a = pos[p.id], b = pos[p.spouseId];
      [-2,2].forEach(off=>{
        const l = document.createElementNS('http://www.w3.org/2000/svg','line');
        l.setAttribute('x1',a.x+22); l.setAttribute('x2',b.x-22);
        l.setAttribute('y1',a.y+off); l.setAttribute('y2',b.y+off);
        l.setAttribute('stroke','#dc2626'); l.setAttribute('stroke-width','1.5');
        svg.appendChild(l);
      });
    }
  });

  // Draw persons
  persons.forEach(p=>{
    const pt = pos[p.id]; if(!pt) return;
    const g = document.createElementNS('http://www.w3.org/2000/svg','g');
    g.setAttribute('transform',`translate(${pt.x},${pt.y})`);
    g.style.cursor='pointer';

    let shape;
    if(p.gender==='male'){
      shape = document.createElementNS('http://www.w3.org/2000/svg','rect');
      shape.setAttribute('x',-22); shape.setAttribute('y',-22);
      shape.setAttribute('width',44); shape.setAttribute('height',44);
    }else{
      shape = document.createElementNS('http://www.w3.org/2000/svg','circle');
      shape.setAttribute('r',22);
    }
    const color = p.role==='caregiver' ? '#fbbf24' : (p.gender==='male' ? '#bfdbfe' : '#fbcfe8');
    shape.setAttribute('fill', p.status==='deceased' ? '#cbd5e1' : color);
    shape.setAttribute('stroke','#475569'); shape.setAttribute('stroke-width','2');
    g.appendChild(shape);

    // Death cross
    if(p.status==='deceased'){
      const x1 = document.createElementNS('http://www.w3.org/2000/svg','line');
      x1.setAttribute('x1',-22); x1.setAttribute('y1',-22);
      x1.setAttribute('x2',22); x1.setAttribute('y2',22);
      x1.setAttribute('stroke','#991b1b'); x1.setAttribute('stroke-width','2');
      g.appendChild(x1);
      const x2 = document.createElementNS('http://www.w3.org/2000/svg','line');
      x2.setAttribute('x1',22); x2.setAttribute('y1',-22);
      x2.setAttribute('x2',-22); x2.setAttribute('y2',22);
      x2.setAttribute('stroke','#991b1b'); x2.setAttribute('stroke-width','2');
      g.appendChild(x2);
    }

    // Caregiver star
    if(p.role==='caregiver'){
      const star = document.createElementNS('http://www.w3.org/2000/svg','text');
      star.setAttribute('x',20); star.setAttribute('y',-15);
      star.setAttribute('font-size','16'); star.textContent='⭐';
      g.appendChild(star);
    }

    // Name label
    const name = document.createElementNS('http://www.w3.org/2000/svg','text');
    name.setAttribute('y',38); name.setAttribute('text-anchor','middle');
    name.setAttribute('font-size','12'); name.setAttribute('font-family','inherit');
    name.setAttribute('font-weight','600'); name.setAttribute('fill','#0f172a');
    name.textContent = p.name.length>14 ? p.name.slice(0,14)+'…' : p.name;
    g.appendChild(name);

    if(p.age!==undefined && p.age!==null && p.age!==''){
      const age = document.createElementNS('http://www.w3.org/2000/svg','text');
      age.setAttribute('y',52); age.setAttribute('text-anchor','middle');
      age.setAttribute('font-size','11'); age.setAttribute('fill','#64748b');
      age.textContent = p.age+' ปี';
      g.appendChild(age);
    }

    g.onclick = ()=>openPersonModal(p);
    svg.appendChild(g);
  });
}

function renderCalendar(){
  const c = currentCommunity();
  if(!c.calendarEvents) c.calendarEvents = {};
  const el = document.getElementById('calendarGrid');
  el.innerHTML = THAI_MONTHS.map((m,i)=>{
    const events = c.calendarEvents[i]||[];
    return `
      <div class="month-card">
        <h4>${m} <button class="btn btn-sm btn-primary btn-icon" onclick="openCalendarModal(${i})">+</button></h4>
        <div class="events">
          ${events.length ? events.map((e,idx)=>`
            <span class="event-tag" title="${escapeHtml(e.desc||'')}">
              ${eventIcon(e.type)} ${escapeHtml(e.title)}
              <span class="x" onclick="removeCalendarEvent(${i},${idx})">×</span>
            </span>
          `).join('') : '<span class="small">— ไม่มีกิจกรรม —</span>'}
        </div>
      </div>`;
  }).join('');
}

function eventIcon(t){
  return {harvest:'🌾',festival:'🎉',migration:'🚌',health:'🏥',weather:'☔',other:'📌'}[t]||'📌';

function saveCalendarEvent(){
  const c = currentCommunity();
  const m = parseInt(document.getElementById('calMonth').value);
  const title = document.getElementById('calTitle').value.trim();
  if(!title){ alert('กรุณากรอกชื่อกิจกรรม'); return; }
  if(!c.calendarEvents) c.calendarEvents={};
  if(!c.calendarEvents[m]) c.calendarEvents[m]=[];
  c.calendarEvents[m].push({
    title,
    type: document.getElementById('calType').value,
    desc: document.getElementById('calDesc').value.trim()
  });
  saveState();
  closeModal('calendarModal');
  renderCalendar();
  toast('เพิ่มกิจกรรมแล้ว');
}

  const c = currentCommunity();

  saveState();
  renderCalendar();

/* ============ Import / Export ============ */
function exportData(){
  const blob = new Blob([JSON.stringify(state, null, 2)], {type:'application/json'});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'community-file-backup-'+new Date().toISOString().slice(0,10)+'.json';
  a.click();
  URL.revokeObjectURL(url);
  toast('ส่งออกข้อมูลแล้ว');
}

function importData(evt){
  const file = evt.target.files[0];
  if(!file) return;
  const r = new FileReader();
  r.onload = (e)=>{
    try{
      const imported = JSON.parse(e.target.result);
      if(!imported.communities) throw new Error('Invalid format');
      if(confirm('นำเข้าข้อมูลจะแทนที่ข้อมูลปัจจุบัน — ต้องการดำเนินการต่อ?')){
        state = imported;
        currentCommunityId = state.communities[0]?.id;
        saveState();
        renderAll();
        toast('นำเข้าข้อมูลสำเร็จ');
      }
    }catch(err){
      alert('ไม่สามารถนำเข้าได้: '+err.message);
    }
  };
  r.readAsText(file);
  evt.target.value='';
}

/* ============ Helpers ============ */
function escapeHtml(s){
  if(s===null||s===undefined) return '';
  return String(s).replace(/[&<>"']/g, c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
}
